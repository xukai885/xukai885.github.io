拉取镜像

harbor-nginx.yaml

apiVersion: apps/v1
kind: Deployment
metadata:
  name: nginx-test
spec:
  replicas: 1
  selector:
    matchLabels:
      app: nginx-test
  template:
    metadata:
      labels:  
        app: nginx-test
    spec:
      containers:
      - name: nginx-test
        image: harbor.test.com/nginx-test:latest # 使用 Harbor 上的镜像
        imagePullPolicy: Always
        ports:
        - containerPort: 80 
      imagePullSecrets:
      - name: harbor-key # 使用刚刚创建的 Harbor Secrets

# run
docker run -d -m 2g --cpus=2 \

-v $PWD/jenkins:/var/jenkins_home \
-v /var/run/docker.sock:/var/run/docker.sock \
-v /bin/docker:/bin/docker \
--name jenkins \
--user root \
